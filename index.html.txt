  <script>
    const $ = s => document.querySelector(s),
          $$ = s => document.querySelectorAll(s),
          msg = (t,c='black')=>{$('#msg').textContent=t;$('#msg').style.color=c;};

    let tailors=[], ops=[];

    function toggleRemove(){
      const rows=$$('tbody tr');
      rows.forEach(r=>r.querySelector('.removeBtn').disabled=rows.length===1);
    }

    function fillRow(tr){
      tr.innerHTML=`
        <td class="op-col">
          <button type="button" class="btn small addRowInline">＋</button>
          <select class="operation form-control"><option value="">--</option></select>
        </td>
        <td class="garment-col"><input class="garment form-control" readonly /></td>
        <td class="rate-col"><input class="rate form-control" readonly /></td>
        <td class="tailor-col"><select class="tailor form-control"><option value="">--</option></select></td>
        <td class="qty-col"><input type="number" min="1" class="quantity form-control" /></td>
        <td class="amt-col"><input class="amount form-control" readonly /></td>
        <td class="action-col"><button type="button" class="removeBtn">×</button></td>`;
      // populate selects
      const os=tr.querySelector('.operation'),
            ts=tr.querySelector('.tailor');
      ops.forEach(o=>os.add(new Option(o,o)));
      tailors.forEach(t=>ts.add(new Option(t,t)));

      // bind events
      tr.querySelector('.removeBtn').addEventListener('click',()=>{
        tr.remove(); toggleRemove();
      });
      tr.querySelector('.operation').addEventListener('change',e=>{
        const v=e.target.value, g=tr.querySelector('.garment'),
              r=tr.querySelector('.rate');
        g.value=r.value=''; tr.querySelector('.quantity').value='';
        tr.querySelector('.amount').value='';
        if(!v) return;
        google.script.run
          .withFailureHandler(er=>msg(er.message,'red'))
          .withSuccessHandler(d=>{
            g.value=d.garment;
            r.value=parseFloat(d.rate).toFixed(2);
          })
          .getOperationDetails(v);
      });
      tr.querySelector('.quantity').addEventListener('input',()=>{
        const rate=parseFloat(tr.querySelector('.rate').value)||0,
              qty =parseFloat(tr.querySelector('.quantity').value)||0;
        tr.querySelector('.amount').value=(rate*qty).toFixed(2);
      });
      tr.querySelector('.addRowInline').addEventListener('click',addRow);
    }

    function addRow(){
      const tr=document.createElement('tr');
      $('tbody').appendChild(tr);
      fillRow(tr);
      toggleRemove();
    }

    document.addEventListener('DOMContentLoaded',()=>{
      google.script.run
        .withFailureHandler(er=>msg(er.message,'red'))
        .withSuccessHandler(list=>{
          const sel=$('#style');
          sel.innerHTML='<option value="">-- Select --</option>';
          list.forEach(v=>sel.add(new Option(v,v)));
        })
        .getStyles();

      google.script.run
        .withFailureHandler(er=>msg(er.message,'red'))
        .withSuccessHandler(list=>tailors=list)
        .getTailors();

      $('#style').addEventListener('change',e=>{
        ops=[];
        google.script.run
          .withFailureHandler(er=>msg(er.message,'red'))
          .withSuccessHandler(list=>{
            ops=list;
            $$('tbody tr').forEach(fillRow);
          })
          .getOperations(e.target.value);
      });

      // one row by default
      addRow();

      $('#entryForm').addEventListener('submit',e=>{
        e.preventDefault();
        const style=$('#style').value, date=$('#globalDate').value;
        if(!style) return msg('Select style','red');
        if(!date)  return msg('Select date','red');

        const trs=$$('tbody tr');
        if(!trs.length) return msg('Add operations','red');

        const rows=[];
        for(const tr of trs){
          const op=tr.querySelector('.operation').value,
                tl=tr.querySelector('.tailor').value,
                qt=tr.querySelector('.quantity').value;
          if(!op||!tl||!qt) return msg('Complete all fields','red');
          rows.push({
            operation:op,
            garment:tr.querySelector('.garment').value,
            rate:tr.querySelector('.rate').value,
            tailor:tl,
            quantity:qt,
            amount:tr.querySelector('.amount').value
          });
        }

        $('#submitBtn').disabled=true;
        msg('Saving…');
        google.script.run
          .withFailureHandler(er=>{
            msg('Error:'+er.message,'red');
            $('#submitBtn').disabled=false;
          })
          .withSuccessHandler(res=>{
            $('#submitBtn').disabled=false;
            if(res.success){
              msg('Saved!','green');
              $('tbody').innerHTML='';
              addRow();
            } else {
              msg(res.message,'red');
            }
          })
          .submitForm({ style, date, rows });
      });
    });
  </script>
